## 2.60
```c
#include <stdio.h>

unsigned replace_type(unsigned x, int i, unsigned char b){
	unsigned mask = (0xFF << (i * 8 ));
	mask = ~mask;
	//	printf("%x\n", mask);
	unsigned value = (unsigned)b << (i*8);
	unsigned y = x & mask;
	y = y | value;
	return y;
}

int main(){
	unsigned y1 = replace_type(0x12345678, 2, 0xAB);
	// 0x12AB5678
	printf("%x", y1);
}
```
